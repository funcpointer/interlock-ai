```python
"""Persistent on-disk cache wrapper.

Wraps diskcache with content-hash-based keys so cache invalidation is
automatic when inputs change. All cached values are JSON-serializable
(Pydantic models use ``.model_dump()`` / ``.model_validate()``).
"""

from __future__ import annotations

import hashlib
import json
from pathlib import Path
from typing import Any

import diskcache

CACHE_DIR = Path("data/cache")
CACHE_DIR.mkdir(parents=True, exist_ok=True)
_cache = diskcache.Cache(str(CACHE_DIR), size_limit=int(2e9))  # 2 GB cap


def _key(namespace: str, payload: dict[str, Any]) -> str:
    """Deterministic key from namespace + sorted-JSON payload."""
    blob = json.dumps(payload, sort_keys=True, separators=(",", ":")).encode()
    return f"{namespace}:{hashlib.sha256(blob).hexdigest()}"


def get_or_compute(namespace: str, payload: dict[str, Any], compute):
    key = _key(namespace, payload)
    hit = _cache.get(key)
    if hit is not None:
        return hit, True
    value = compute()
    _cache.set(key, value)
    return value, False


def stats() -> dict[str, int]:
    return {"hits": _cache.stats(enable=False)[0], "misses": _cache.stats(enable=False)[1]}


def clear_namespace(namespace: str) -> int:
    """Drop all keys with the given namespace prefix. Returns count cleared."""
    n = 0
    for k in list(_cache):
        if isinstance(k, str) and k.startswith(f"{namespace}:"):
            del _cache[k]
            n += 1
    return n
```

```python
# src/interlock/detect/tolerances.py
"""Per-attribute engineering tolerance bands.

These values establish "what counts as a meaningful deviation" per parameter
family. Production deployments will replace this with project-specific
tolerance configs. Sources cited inline.
"""

from __future__ import annotations
from dataclasses import dataclass
from typing import Literal

Severity = Literal["critical", "major", "minor", "info"]


@dataclass(frozen=True)
class ToleranceBand:
    attribute_family: str
    rel_tolerance_pct: float       # below this → "info" or "minor"
    rel_major_pct: float           # above this → "major"
    rel_critical_pct: float        # above this → "critical"
    source: str


# Sources:
# - IEEE C57.12.00 Table 17 (transformer impedance tolerance ±7.5 % typical)
# - IEC 60076-1 §5.3 (voltage ratio tolerance ±0.5 % or 1/10 of impedance)
# - NEMA TR-1 (US transformer standards)
# - Industry common practice where standards underspecify (flagged below)

TOLERANCE_TABLE: dict[str, ToleranceBand] = {
    "impedance_pct": ToleranceBand(
        attribute_family="impedance_pct",
        rel_tolerance_pct=7.5,
        rel_major_pct=10.0,
        rel_critical_pct=50.0,
        source="IEEE C57.12.00 §9.1, Table 17",
    ),
    "rated_power_kva": ToleranceBand(
        attribute_family="rated_power_kva",
        rel_tolerance_pct=5.0,
        rel_major_pct=10.0,
        rel_critical_pct=50.0,
        source="IEEE C57.12.00 §5.10",
    ),
    "voltage_kv": ToleranceBand(
        attribute_family="voltage_kv",
        rel_tolerance_pct=0.5,
        rel_major_pct=5.0,
        rel_critical_pct=50.0,
        source="IEC 60076-1 §5.3, IEEE C57.12.00 §5.7",
    ),
    "fault_current_a": ToleranceBand(
        attribute_family="fault_current_a",
        rel_tolerance_pct=5.0,
        rel_major_pct=20.0,
        rel_critical_pct=50.0,
        source="Industry typical; see BACKLOG.md (Phase 14 standards-as-authority)",
    ),
}


def classify(attribute_family: str, deviation_pct: float) -> Severity:
    """Bucket a relative deviation into a severity tier.

    Decimal-shift errors (deviation ≥ 50 %) always classify as critical.
    Below the family's typical tolerance: info. Between: minor or major.
    """
    band = TOLERANCE_TABLE.get(attribute_family)
    if band is None:
        # Unknown family — fall back to broad bands.
        band = ToleranceBand("unknown", 1.0, 5.0, 50.0, "default fallback")
    if deviation_pct >= band.rel_critical_pct:
        return "critical"
    if deviation_pct >= band.rel_major_pct:
        return "major"
    if deviation_pct >= band.rel_tolerance_pct:
        return "minor"
    return "info"


def relative_deviation(a: float, b: float) -> float:
    if a == 0 and b == 0:
        return 0.0
    base = max(abs(a), abs(b))
    return abs(a - b) / base * 100.0
```
